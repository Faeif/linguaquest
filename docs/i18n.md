# i18n & Multi-language Architecture — LinguaQuest
# Version: 1.0.0

## 🎯 Design Principles

```
1. Language as Config
   เพิ่มภาษาใหม่ = เพิ่ม config file เท่านั้น
   ไม่ต้อง rewrite core code

2. Separation of Concerns
   UI Language (interface_lang) แยกจาก
   Learning Language (target_lang) แยกจาก
   Native Language (native_lang)

3. Graceful Fallback
   ถ้าไม่มี translation → fallback เป็น EN
   ถ้าไม่มี IPA → ไม่แสดง ไม่ error

4. Script-aware Rendering
   Latin / CJK / Arabic / Devanagari / Hangul
   แต่ละ script มี renderer ของตัวเอง

5. RTL-ready ตั้งแต่ต้น
   ใช้ CSS logical properties ทุกที่
   dir="rtl" บน <html> เพียงพอ
```

### cards table — Multi-language ready
```sql
CREATE TABLE cards (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  deck_id UUID REFERENCES decks(id) ON DELETE CASCADE,

  -- Core
  word TEXT NOT NULL,
  target_lang TEXT NOT NULL DEFAULT 'en',

  -- Script variants (for CJK/Arabic)
  native_script TEXT,          -- Kanji: 曖昧, Arabic: غامض
  romanization TEXT,           -- Romaji: aimai, Pinyin: mó hú
  phonetic TEXT,               -- IPA or language-specific phonetics

  -- Audio
  audio_url TEXT,              -- Primary pronunciation
  audio_url_alt TEXT,          -- Alt pronunciation (US/UK, etc.)

  -- Meanings — JSONB supports all languages
  meanings JSONB DEFAULT '{}',
  -- { "th": "คลุมเครือ", "vi": "mơ hồ", "id": "ambigu", "en": "unclear" }

  -- Examples
  examples JSONB DEFAULT '[]',
  -- [{
  --   "text": "The contract was ambiguous.",
  --   "phonetic": null,
  --   "translations": { "th": "สัญญานั้นคลุมเครือ", "vi": "..." }
  -- }]

  -- Language-specific metadata
  lang_metadata JSONB DEFAULT '{}',
  -- JA: { "kanji": "曖昧", "hiragana": "あいまい", "jlpt": "N2",
  --       "pitch_accent": "2", "stroke_count": 15 }
  -- ZH: { "simplified": "模糊", "traditional": "模糊",
  --       "pinyin": "mó hú", "hsk": "5", "tones": "2 2" }
  -- KO: { "hangul": "모호한", "romanization": "mo-ho-han",
  --       "topik": "intermediate" }
  -- AR: { "root": "غمض", "pattern": "فاعل",
  --       "diacritics": "غَامِض" }

  -- Proficiency level (system varies by language)
  proficiency_level TEXT,
  -- EN/VI/ID: 'A1'-'C2' (CEFR)
  -- JA: 'N5'-'N1' (JLPT)
  -- ZH: 'HSK1'-'HSK6'
  -- KO: 'TOPIK1'-'TOPIK6'

  -- Legacy (keep for backward compat)
  cefr_level TEXT,
  meaning_th TEXT,             -- Kept for TH users, synced with meanings.th

  position INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Index สำหรับ query by language
CREATE INDEX idx_cards_target_lang ON cards(target_lang);
CREATE INDEX idx_cards_deck_lang ON cards(deck_id, target_lang);

-- Trigger: sync meaning_th กับ meanings.th
CREATE OR REPLACE FUNCTION sync_meaning_th()
RETURNS TRIGGER AS $$
BEGIN
  IF NEW.meanings ? 'th' THEN
    NEW.meaning_th := NEW.meanings->>'th';
  ELSIF NEW.meaning_th IS NOT NULL THEN
    NEW.meanings := NEW.meanings || jsonb_build_object('th', NEW.meaning_th);
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER sync_meaning_th_trigger
  BEFORE INSERT OR UPDATE ON cards
  FOR EACH ROW EXECUTE FUNCTION sync_meaning_th();
```

### Universal Card Generator
```typescript
// packages/core/ai/card-generator.ts

export async function generateCard(
  word: string,
  targetLang: string,
  nativeLang: string,
  additionalLangs: string[] = []
): Promise<GeneratedCard> {

  const langConfig = getLanguage(targetLang)
  const cacheKey = `ai:card:${targetLang}:${word}:${nativeLang}`

  // Check cache first
  const cached = await redis.get<GeneratedCard>(cacheKey)
  if (cached) return { ...cached, cached: true }

  // Build meanings request for all needed languages
  const meaningLangs = [nativeLang, 'en', ...additionalLangs].filter(
    (l, i, arr) => arr.indexOf(l) === i // unique
  )

  const prompt = buildCardPrompt(word, langConfig, meaningLangs)
  const result = await geminiFlash.generateContent(prompt)
  const card = parseCardResponse(result, langConfig)

  // Cache 30 days
  await redis.setex(cacheKey, 60 * 60 * 24 * 30, JSON.stringify(card))

  return card
}

function buildCardPrompt(
  word: string,
  lang: LanguageConfig,
  meaningLangs: string[]
): string {

  const langNames: Record<string, string> = {
    th: 'Thai', vi: 'Vietnamese', id: 'Indonesian',
    en: 'English', ja: 'Japanese', ko: 'Korean',
    zh: 'Chinese', ar: 'Arabic',
  }

  const meaningsRequest = meaningLangs
    .map(l => `"${l}": "meaning in ${langNames[l] ?? l}"`)
    .join(',\n    ')

  let specialInstructions = ''

  if (lang.specialHandling.includes('furigana')) {
    specialInstructions += `
For Japanese words, include in lang_metadata:
- "hiragana": hiragana reading
- "kanji": kanji form (if different from word)
- "jlpt": JLPT level (N1-N5)
- "pitch_accent": pitch accent number`
  }

  if (lang.specialHandling.includes('rtl')) {
    specialInstructions += `
For Arabic words, include in lang_metadata:
- "root": trilateral root
- "pattern": morphological pattern
- "diacritics": fully voweled form`
  }

  if (lang.hasTones) {
    specialInstructions += `
Include tone information in the phonetic field.`
  }

  return `Generate a vocabulary flashcard for the ${lang.geminiName} word: "${word}"

Return ONLY valid JSON (no markdown, no explanation):
{
  "word": "${word}",
  "phonetic": ${lang.hasIPA ? '"IPA notation"' : lang.hasRomanization ? '"romanization"' : 'null'},
  "romanization": ${lang.hasRomanization ? '"romanization/transliteration"' : 'null'},
  "meanings": {
    ${meaningsRequest}
  },
  "examples": [
    {
      "text": "Natural example sentence in ${lang.geminiName}",
      "translations": {
        ${meaningLangs.map(l => `"${l}": "translation in ${langNames[l] ?? l}"`).join(',\n        ')}
      }
    },
    {
      "text": "Second example sentence",
      "translations": {
        ${meaningLangs.map(l => `"${l}": "translation"`).join(',\n        ')}
      }
    }
  ],
  "collocations": ["common phrase 1", "common phrase 2", "common phrase 3"],
  "proficiency_level": "${lang.proficiencyLevels[2] ?? 'B1'}",
  "register": "formal|informal|academic|technical|neutral",
  "lang_metadata": {}
}
${specialInstructions}`
}
```

export async function generateFonts(langs: string[]) {
  const fonts = []

  if (langs.some(l => ['ja'].includes(l))) {
    fonts.push(Noto_Sans_JP({ subsets: ['latin'], display: 'swap' }))

### CardRenderer — Universal
```typescript
// components/features/study/CardRenderer.tsx

export function CardRenderer({
  card,
  userNativeLang,
  showMeaning = false
}: CardRendererProps) {

  const langConfig = getLanguage(card.target_lang)
  const meaning = card.meanings[userNativeLang]
    ?? card.meanings['en']
    ?? card.meaning_th

  return (
    <div
      dir={langConfig.direction}
      lang={card.target_lang}
      className="flex flex-col gap-3"
    >
      {/* Main word — script-aware sizing */}
      <div
        className="text-center"
        style={{
          fontSize: `${langConfig.fontSize.card * 2.5}rem`,
          fontFamily: langConfig.fontFamily,
          fontWeight: 500,
        }}
      >
        {/* Show native script if available */}
        {card.native_script ?? card.word}
      </div>

      {/* Furigana for Japanese */}
      {card.lang_metadata?.hiragana && (
        <div className="text-center text-sm text-secondary">
          {card.lang_metadata.hiragana}
        </div>
      )}

      {/* Romanization / IPA */}
      {card.romanization && (
        <div className="text-center text-sm text-secondary font-mono">
          {card.romanization}
        </div>
      )}
      {card.phonetic && !card.romanization && (
        <div className="text-center text-sm text-secondary font-mono">
          /{card.phonetic}/
        </div>
      )}

      {/* Meaning — shown on flip */}
      {showMeaning && (
        <div
          dir="ltr"  // Meaning always LTR (in user's native lang)
          className="text-center text-lg mt-4"
        >
          {meaning}
        </div>
      )}

      {/* Examples */}
      {showMeaning && card.examples.map((ex, i) => (
        <div key={i} className="text-sm">
          <div
            dir={langConfig.direction}
            style={{ fontFamily: langConfig.fontFamily }}
            className="text-secondary italic"
          >
            {ex.text}
          </div>
          <div dir="ltr" className="text-tertiary text-xs mt-1">
            {ex.translations[userNativeLang] ?? ex.translations['en']}
          </div>
        </div>
      ))}
    </div>
  )
}
```

### CSS — Logical Properties (RTL-safe)
```css
/* ✅ Use logical properties — works with both LTR and RTL */
.card { padding-inline: 1rem; }        /* = padding-left + padding-right */
.icon { margin-inline-end: 0.5rem; }   /* = margin-right in LTR, margin-left in RTL */
.sidebar { border-inline-end: 1px; }   /* = border-right in LTR */

/* ✅ Tailwind equivalents */
/* ps-4 = padding-inline-start (not pl-4) */
/* me-2 = margin-inline-end (not mr-2) */
/* border-e = border-inline-end (not border-r) */

/* ❌ Avoid directional properties */
/* padding-left, margin-right, border-right */
/* pl-4, mr-2, border-r */
```
